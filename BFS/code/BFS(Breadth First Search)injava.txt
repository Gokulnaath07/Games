public void BFS(int v){
	int totalVertex=adjList.size();//total no of vertices
	boolean[] visited=new boolean[totalVertex];
	
	visited[v]=true;

	queue<Integer> q=new LinkedList<>();
	q.add(v);
	
	while(q.size()!=0){
		int vertex=q.remove();
		for (int i=0; adjList.get(vertex).size();i++){
			int av=adjList.get(vertex).get(i);
			if(!visited[av]){
				q.add(av);
				visited[av]=true;
			}
		}
	}
}
